CREATE TABLE Estado (
    uf VARCHAR(2) PRIMARY KEY,
    nome VARCHAR(20)
);


CREATE TABLE Cidade (
    id_cidade INTEGER PRIMARY KEY,
    uf VARCHAR(2),
    nome VARCHAR(75)
);


CREATE TABLE Bairro (
    id_bairro INTEGER PRIMARY KEY,
    id_cidade INTEGER,
    nome VARCHAR(75),
    CONSTRAINT fk_bairro_cidade FOREIGN KEY (id_cidade) REFERENCES Cidade(id_cidade)
);


CREATE TABLE TipoEndereco (
    id_tipo_endereco INTEGER PRIMARY KEY,
    tipo VARCHAR(30)
);


CREATE TABLE Endereco (
    id_endereco INTEGER PRIMARY KEY,
    id_tipo_endereco INTEGER,
    id_bairro INTEGER,
    nome VARCHAR(100),
    numero VARCHAR(10),
    cep VARCHAR(10),
    CONSTRAINT fk_endereco_tipo FOREIGN KEY (id_tipo_endereco) REFERENCES TipoEndereco(id_tipo_endereco),
    CONSTRAINT fk_endereco_bairro FOREIGN KEY (id_bairro) REFERENCES Bairro(id_bairro)
);


CREATE TABLE Fornecedor (
    id_fornecedor INTEGER PRIMARY KEY,
    id_endereco INTEGER,
    nome VARCHAR(45),
    CONSTRAINT fk_fornecedor_endereco FOREIGN KEY (id_endereco) REFERENCES Endereco(id_endereco)
);


CREATE TABLE Categoria (
    id_categoria INTEGER PRIMARY KEY,
    tipo VARCHAR(45)
);


CREATE TABLE Produto (
    id_produto INTEGER PRIMARY KEY,
    id_fornecedor INTEGER,
    id_categoria INTEGER,
    nome VARCHAR(45),
    quantidade INTEGER,
    valor_unitario NUMERIC(5,2),
    data_validade DATE,
    CONSTRAINT fk_produto_fornecedor FOREIGN KEY (id_fornecedor) REFERENCES Fornecedor(id_fornecedor),
    CONSTRAINT fk_produto_categoria FOREIGN KEY (id_categoria) REFERENCES Categoria(id_categoria)
);


CREATE TABLE Cliente (
    id_cliente INTEGER PRIMARY KEY,
    id_endereco INTEGER,
    nome VARCHAR(75),
    sexo VARCHAR(1),
    data_nascimento DATE,
    CONSTRAINT fk_cliente_endereco FOREIGN KEY (id_endereco) REFERENCES Endereco(id_endereco)
);


CREATE TABLE Departamento (
    id_departamento INTEGER PRIMARY KEY,
    nome VARCHAR(45)
);


CREATE TABLE Supervisor (
    id_supervisor INTEGER PRIMARY KEY,
    id_departamento INTEGER,
    nome VARCHAR(75),
    CONSTRAINT fk_supervisor_departamento FOREIGN KEY (id_departamento) REFERENCES Departamento(id_departamento)
);


CREATE TABLE Funcionario (
    id_funcionario INTEGER PRIMARY KEY,
    id_endereco INTEGER,
    id_supervisor INTEGER,
    nome VARCHAR(50),
    data_nascimento DATE,
    sexo VARCHAR(1),
    data_contratacao DATE,
    data_demissao DATE,
    demissao VARCHAR(2),
    salario NUMERIC(5,2),
    CONSTRAINT fk_funcionario_endereco FOREIGN KEY (id_endereco) REFERENCES Endereco(id_endereco),
    CONSTRAINT fk_funcionario_supervisor FOREIGN KEY (id_supervisor) REFERENCES Supervisor(id_supervisor)
);


CREATE TABLE Dependente (
    id_dependente INTEGER PRIMARY KEY,
    id_funcionario INTEGER,
    nome VARCHAR(50),
    parentesco VARCHAR(45),
    sexo VARCHAR(1),
    data_nascimento DATE,
    CONSTRAINT fk_dependente_funcionario FOREIGN KEY (id_funcionario) REFERENCES Funcionario(id_funcionario)
);


CREATE TABLE Departamento_Funcionario (
    id_depart_func INTEGER PRIMARY KEY,
    id_funcionario INTEGER,
    id_departamento INTEGER,
    data_entrada DATE,
    data_saida DATE,
    CONSTRAINT fk_depart_func_funcionario FOREIGN KEY (id_funcionario) REFERENCES Funcionario(id_funcionario),
    CONSTRAINT fk_depart_func_departamento FOREIGN KEY (id_departamento) REFERENCES Departamento(id_departamento)
);


CREATE TABLE Telefone (
    id_telefone INTEGER PRIMARY KEY,
    id_cliente INTEGER,
    id_funcionario INTEGER,
    numero VARCHAR(14),
    CONSTRAINT fk_telefone_cliente FOREIGN KEY (id_cliente) REFERENCES Cliente(id_cliente),
    CONSTRAINT fk_telefone_funcionario FOREIGN KEY (id_funcionario) REFERENCES Funcionario(id_funcionario)
);


CREATE TABLE Compra (
    id_compra INTEGER PRIMARY KEY,
    id_cliente INTEGER,
    id_funcionario INTEGER,
    data_compra DATE,
    valor NUMERIC(5,2),
    forma_pagamento VARCHAR(15),
    finalizada VARCHAR(1),
    CONSTRAINT fk_compra_cliente FOREIGN KEY (id_cliente) REFERENCES Cliente(id_cliente),
    CONSTRAINT fk_compra_funcionario FOREIGN KEY (id_funcionario) REFERENCES Funcionario(id_funcionario)
);


CREATE TABLE Item_Compra (
    id_compra INTEGER,
    id_produto INTEGER,
    quantidade INTEGER,
    valor_total NUMERIC,
    PRIMARY KEY (id_compra, id_produto),
    CONSTRAINT fk_item_compra FOREIGN KEY (id_compra) REFERENCES Compra(id_compra),
    CONSTRAINT fk_item_produto FOREIGN KEY (id_produto) REFERENCES Produto(id_produto)
);
